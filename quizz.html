                } else {

                    showResult();

                }

            });


            // Gère le clic sur "Recommencer"

            restartBtn.addEventListener('click', () => {

                hideResult();

                showQuiz();

                currentQuestionIndex = 0;

                score = 0;

                displayQuestion();

            });


            // Gère la fermeture de la modale

            closeQuizModalBtn.addEventListener('click', () => {

                quizModal.classList.remove('active');

            });

            quizModal.addEventListener('click', (e) => {

                if (e.target === quizModal) {

                    quizModal.classList.remove('active');

                }

            });


            function showQuiz() {

                modalTitle.textContent = `Quiz ${quizLevel}`;

                document.getElementById('quizContent').classList.remove('hidden');

                document.getElementById('quizResult').classList.add('hidden');

                feedback.classList.add('hidden');

                nextQuestionBtn.classList.add('hidden');

                quizModal.classList.add('active');

            }


            function displayQuestion() {

                const questionData = currentQuiz[currentQuestionIndex];

                questionText.textContent = questionData.question;

                optionsContainer.innerHTML = '';

                feedback.classList.add('hidden');

                nextQuestionBtn.classList.add('hidden');


                questionData.options.forEach((option, index) => {

                    const button = document.createElement('button');

                    button.textContent = option;

                    button.className = 'choice-btn p-4 rounded-lg shadow-md hover:bg-fcd34d transition duration-200 text-left';

                    button.addEventListener('click', () => checkAnswer(index, questionData));

                    optionsContainer.appendChild(button);

                });

            }


            function checkAnswer(selectedIndex, questionData) {

                const allButtons = optionsContainer.querySelectorAll('.choice-btn');

                const isCorrect = Array.isArray(questionData.correctAnswerIndex) ? 

                                  questionData.correctAnswerIndex.includes(selectedIndex) : 

                                  selectedIndex === questionData.correctAnswerIndex;


                if (isCorrect) {

                    score++;

                }


                allButtons.forEach((button, index) => {

                    if (Array.isArray(questionData.correctAnswerIndex)) {

                        if (questionData.correctAnswerIndex.includes(index)) {

                            button.classList.add('correct');

                        } else {

                            button.classList.add('bg-gray-200');

                        }

                    } else {

                        if (index === questionData.correctAnswerIndex) {

                            button.classList.add('correct');

                        } else if (index === selectedIndex) {

                            button.classList.add('incorrect');

                        } else {

                            button.classList.add('bg-gray-200');

                        }

                    }

                    button.disabled = true; // Empêche de cliquer après la réponse

                });


                showFeedback(isCorrect, questionData.explanation);

                nextQuestionBtn.classList.remove('hidden');

            }


            function showFeedback(isCorrect, explanation) {

                feedback.classList.remove('hidden');

                feedback.innerHTML = `<p class="font-bold">${isCorrect ? 'Correct !' : 'Incorrect !'}</p><p>${explanation}</p>`;

                if (isCorrect) {

                    feedback.classList.add('bg-green-100', 'text-green-800');

                    feedback.classList.remove('bg-red-100', 'text-red-800');

                } else {

                    feedback.classList.add('bg-red-100', 'text-red-800');

                    feedback.classList.remove('bg-green-100', 'text-green-800');

                }

            }


            function showResult() {

                document.getElementById('quizContent').classList.add('hidden');

                document.getElementById('quizResult').classList.remove('hidden');

                scoreText.textContent = `Votre score est de ${score} sur ${currentQuiz.length}.`;

            }


            function hideResult() {

                document.getElementById('quizResult').classList.add('hidden');

            }

        });
